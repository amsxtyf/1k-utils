mergeToUrl

Types

function default(url: string, query: Object): string

Usage

const url = 'https://www.google.cn'

mergeToUrl(url, { key: 'value' })
mergeToUrl(url + '?', { key: 'value' })
mergeToUrl(url + '?', { key: 'value' })
// => 'https://www.google.cn?key=value'
mergeToUrl(url + '?a=1', { key: 'value' })
// => 'https://www.google.cn?a=1&key=value'
// Don't do that
// ----------------------------------------------------------------------
mergeToUrl(url + '&', { key: 'value' })
